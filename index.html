  $mdThemingProvider.theme('default')
    .primaryPalette('pink')
    .accentPalette('orange');
});

Employee Alert! 


  Employee Alert!

						

var alert;
$scope.showAlert = showAlert;

function showAlert() {
		alert = $mdDialog.alert({
			title: 'Attention',
			content: 'This is an example of how easy dialogs can be!',
			ok: 'Close'
		});
		$mdDialog
			.show( alert )
			.finally(function() {
				alert = undefined;
			});
	}
